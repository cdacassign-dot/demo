1. Install Docker
Install Docker on Ubuntu/Linux system.
Verify Docker installation.


sudo apt update
sudo apt install docker.io -y
sudo systemctl start docker
sudo systemctl enable docker



2. Run Ubuntu Container
Pull Ubuntu image from Docker Hub.
Run container in interactive mode.
Create a directory named cdac .


# On your Ubuntu system:
sudo docker pull ubuntu
sudo docker images
sudo docker run -it ubuntu bash


# Inside the container:
mkdir cdac
ls
exit 



3. Containerize a Java Application
Create a file inside it.
Write a simple Java program.
Create a Dockerfile for the application.
Build Docker image.
Run container from the image.


# Setup
mkdir ~/myapp && cd ~/myapp


# Create files (use nano as shown above)
nano HelloWorld.java
nano Dockerfile


FROM eclipse-temurin:17-jdk-jammy


WORKDIR /app


COPY HelloWorld.java .


RUN javac HelloWorld.java


CMD ["java", "HelloWorld"]


# Build image
sudo docker build -t hellojava .


# Verify image
sudo docker images


# Run container
sudo docker run hellojava





4. Docker Compose
Create:
One Java container
One MySQL container
Run both containers using Docker Compose.




mkdir ~/compose-app && cd ~/compose-app
nano HelloMySQL.java       # paste Java code


public class HelloMySQL {
    public static void main(String[] args) throws InterruptedException {
        System.out.println("=================================");
        System.out.println("  Java Container is running!");
        System.out.println("  Connecting to MySQL container...");
        System.out.println("  MySQL Host: mysql-db:3306");
        System.out.println("  Database: cdacdb");
        System.out.println("  Created by: Utkarsh Kumar");
        System.out.println("=================================");
        Thread.sleep(5000); // keep container alive for 5 seconds
    }
}



nano Dockerfile            # paste Dockerfile


FROM eclipse-temurin:17-jdk-jammy


WORKDIR /app


COPY HelloMySQL.java .


RUN javac HelloMySQL.java


CMD ["java", "HelloMySQL"]




nano docker-compose.yml    # paste compose config


version: '3.8'


services:


  java-app:
    build: .
    container_name: java-container
    depends_on:
      - mysql-db
    environment:
      - DB_HOST=mysql-db
      - DB_PORT=3306


  mysql-db:
    image: mysql:8.0
    container_name: mysql-container
    environment:
      MYSQL_ROOT_PASSWORD: root123
      MYSQL_DATABASE: cdacdb
      MYSQL_USER: utkarsh
      MYSQL_PASSWORD: cdac123
    ports:
      - "3306:3306"




sudo docker compose up     # run both containers
sudo docker compose down   # stop everything





5. Docker Networking
Create a custom Docker network.
Run two containers on the same network.
Verify communication between containers.




# Create network
sudo docker network create cdac-network
sudo docker network ls


# Terminal 1 — container1 (listener)
sudo docker run -it --rm --name container1 --network cdac-network ubuntu bash
# Inside container1:
apt-get update -y && apt-get install -y netcat-openbsd
nc -lp 1234


# Terminal 2 — container2 (sender)


sudo docker run -it --rm --name container2 --network cdac-network ubuntu bash
# Inside container2:
apt-get update -y && apt-get install -y netcat-openbsd
nc container1 1234


# Inspect network
sudo docker network inspect cdac-network


# Cleanup
sudo docker network rm cdac-network




6. Docker Volumes
Create a Docker volume.
Attach volume to a container.
Store files inside mounted directory.
Restart container and verify data persistence.




# Create volume
sudo docker volume create cdac-volume
sudo docker volume ls
sudo docker volume inspect cdac-volume


# Run container1 with volume
sudo docker run -it --name container1 -v cdac-volume:/mydata ubuntu bash
# Inside container:
cd /mydata
echo "Hello from CDAC!" > cdac.txt
echo "Created by: Utkarsh Kumar" > student.txt
cat cdac.txt
exit


# Restart same container → data still there
sudo docker start -i container1
ls /mydata
exit


# Delete container1 and test with new container2
sudo docker rm container1
sudo docker run -it --name container2 -v cdac-volume:/mydata ubuntu bash
ls /mydata       # Files still exist!
cat cdac.txt
exit


# Cleanup
sudo docker rm container2
sudo docker volume rm cdac-volume



7. Docker Hub
Create Docker Hub repository.
Tag Docker image.
Push image to Docker Hub.
Pull image and run container again.




# Login
sudo docker logout
sudo docker login
# Username: cdacassign3000
# Password: <your password or access token>


# Tag image
sudo docker tag hellojava:latest cdacassign3000/hellojava:v1.0


# Verify
sudo docker images


# Push to Docker Hub
sudo docker push cdacassign3000/hellojava:v1.0


# Delete local image
sudo docker rmi cdacassign3000/hellojava:v1.0
sudo docker rmi hellojava:latest


# Pull from Docker Hub
sudo docker pull cdacassign3000/hellojava:v1.0


# Run from pulled image
sudo docker run cdacassign3000/hellojava:v1.0
